// insertarAgenda.php
function insertarDatos($Nombre, $Telefono_oficina, $Telefono_casa, $Celular, $Empresa, $Cargo, $Fecha_cumpleanos, $conn) {
    $sql = "INSERT INTO contactos (Nombre, Telefono_oficina, Telefono_casa, Celular, Empresa, Cargo, Fecha_de_cumpleanos)
                VALUES (:nombre, :telefono_oficina, :telefono_casa, :celular, :empresa, :cargo, :fecha_cumpleanos)";

    try{
        $stmt = $conn->prepare($sql);
        // Asociar parámetro con los valores
        $stmt->bindParam(':nombre', $Nombre, PDO::PARAM_STR);
        $stmt->bindParam(':telefono_oficina', $Telefono_oficina, PDO::PARAM_STR);
        $stmt->bindParam(':telefono_casa', $Telefono_casa, PDO::PARAM_STR);
        $stmt->bindParam(':celular', $Celular, PDO::PARAM_STR);
        $stmt->bindParam(':empresa', $Empresa, PDO::PARAM_STR);
        $stmt->bindParam(':cargo', $Cargo, PDO::PARAM_STR);
        $stmt->bindParam(':fecha_cumpleanos', $Fecha_cumpleanos, PDO::PARAM_STR);

        if($stmt->execute()){
            return "Datos insertados correctamente.";
        } else {
            return "Error al insertar datos";
      }
    } catch (PDOException $e){
        return "Error en la preparación de la consulta: " . $e->getMessage();
    }
}
